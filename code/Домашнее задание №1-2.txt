#Написать “калькулятор”. Пользователь вводит либо числа, либо знаки арифметических действий (+, -, *, /), калькулятор выводит промежуточные ответы.
#Первый ввод обязательно число.
#Приоритезации операций нет, скобки не обрабатываются.
#Окончание работы является ввод пустой строки (т.е. просто нажатие Enter)
#Пример:
#Введите действие или число:
#4
#Введите следующее действие или число:
#+
#Введите следующее действие или число:
#4
#Промежуточный итог: 8
#Введите следующее действие или число:
#*
#Введите следующее действие или число:
#2
#Промежуточный итог: 16
#Введите следующее действие или число:
#Итоговый результат: 16

while True:
    a = input('Введите число: ')
    res = int(a)
    while a:
        print ('Введите следующее действие или число: ')
        operation = input ()
        if operation =='': 
            print ('Итоговый результат:' , res)    
            break
        if operation == '+':
            print ('Введите следующее действие или число:')
            b = int (input ())
            res += b
            print ('Промежуточный итог: ', res)
        elif operation == '-':
            print ('Введите следующее действие или число:')
            b = int (input ())
            res -= b
            print ('Промежуточный итог: ', res)
        elif operation == '*':
            print ('Введите следующее действие или число:')
            b = int (input ())
            res *= b
            print ('Промежуточный итог: ', res)
        elif operation == '/':
            print ('Введите следующее действие или число:')
            b = int (input ())
            if b == 0:
                print ('Ошибка, делить на ноль нельзя!')
                break
            else:
                res /= b
                print ('Промежуточный итог: ', res)
        else:
            print ('Итоговый результат:' , res)
            break
    break
